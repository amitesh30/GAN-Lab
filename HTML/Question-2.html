import os
import torch

from torchvision import transforms
from PIL import Image
import torch.nn.functional as F

# Define the mapping from string labels to integers
label_mapping = {'Cardiomegaly': 0, 'Aortic enlargement': 1}

# Custom dataset class
class CustomDataset(Dataset):
    def __init__(self, csv_file, root_dir, transform=None):
        self.annotations = pd.read_csv(csv_file)
        self.root_dir = root_dir
        self.transform = transform

    def __len__(self):
        return len(self.annotations)

    def __getitem__(self, index):
        img_id = self.annotations.iloc[index, 0]
        img_path = os.path.join(self.root_dir, img_id + '.jpg')
        image = Image.open(img_path).convert('L')  # Convert to grayscale
        original_label = self.annotations.iloc[index, 1]
        label = label_mapping[original_label]  # Convert string label to integer using the mapping
        if self.transform:
            image = self.transform(image)
        
        label_tensor = torch.tensor(label, dtype=torch.long)
        # Return image and one-hot encoded labels
        return image, F.one_hot(label_tensor, num_classes=len(label_mapping)).float()

# Define the transformations, including resizing and normalization
transform = transforms.Compose([
    transforms.Resize((128, 128)),  # Resize the images to 128x128
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.5], std=[0.5])  # Normalize grayscale images
])

import torch
import torch.nn as nn
import torch.nn.functional as F
import torch.optim as optim

# Define the CVAE model

import torch
import torch.nn as nn
import torch.nn.functional as F

# Define the CVAE Encoder
class CVAEEncoder(nn.Module):
    def __init__(self, image_channels, label_dim, hidden_dims, latent_dim):
        super(CVAEEncoder, self).__init__()
        self.fc1 = nn.Linear(image_channels + label_dim, hidden_dims)
        self.fc_mu = nn.Linear(hidden_dims, latent_dim)
        self.fc_var = nn.Linear(hidden_dims, latent_dim)
        
    def forward(self, x, labels):
        # Flatten image and concatenate with the label
        x = torch.flatten(x, start_dim=1)
        x = torch.cat([x, labels], dim=1)
        hidden = F.relu(self.fc1(x))
        mu = self.fc_mu(hidden)
        log_var = self.fc_var(hidden)
        return mu, log_var

# Define the CVAE Decoder
class CVAEDecoder(nn.Module):
    def __init__(self, latent_dim, label_dim, hidden_dims, output_dim):
        super(CVAEDecoder, self).__init__()
        self.fc1 = nn.Linear(latent_dim + label_dim, hidden_dims)
        self.fc2 = nn.Linear(hidden_dims, output_dim)
        
    def forward(self, z, labels):
        z = torch.cat([z, labels], dim=1)
        hidden = F.relu(self.fc1(z))
        reconstruction = torch.sigmoid(self.fc2(hidden))
        reconstruction = reconstruction.view(-1, 1, 128, 128)  # Reshape back to image dimensions
        return reconstruction

# Define the full CVAE
class CVAE(nn.Module):
    def __init__(self, image_channels, label_dim, hidden_dims, latent_dim, output_dim):
        super(CVAE, self).__init__()
        self.encoder = CVAEEncoder(image_channels, label_dim, hidden_dims, latent_dim)
        self.decoder = CVAEDecoder(latent_dim, label_dim, hidden_dims, output_dim)

    def reparameterize(self, mu, log_var):
        std = torch.exp(0.5 * log_var)
        eps = torch.randn_like(std)
        return mu + eps * std

    def forward(self, x, labels):
        mu, log_var = self.encoder(x, labels)
        z = self.reparameterize(mu, log_var)
        return self.decoder(z, labels), mu, log_var

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# Initialize the CVAE model
cvae = CVAE(image_channels=128*128, label_dim=2, hidden_dims=400, latent_dim=20, output_dim=128*128)
cvae = cvae.to(device)

# Define the loss function and optimizer
def cvae_loss(recon_x, x, mu, log_var):
    BCE = F.binary_cross_entropy(recon_x.view(-1, 128*128), x.view(-1, 128*128), reduction='sum')
    KLD = -0.5 * torch.sum(1 + log_var - mu.pow(2) - log_var.exp())
    return BCE + KLD

optimizer = optim.Adam(cvae.parameters(), lr=3e-4)

# Implementing the Training Loop
num_epochs = 1  # Adjust as needed

for epoch in range(num_epochs):
    cvae.train()  # Set the model to training mode
    train_loss = 0
    for batch_idx, (images, labels) in enumerate(train_loader):
        images, labels = images.to(device), labels.to(device)
        optimizer.zero_grad()
        recon_images, mu, log_var = cvae(images, labels)
        loss = cvae_loss(recon_images, images, mu, log_var)
        loss.backward()
        train_loss += loss.item()
        optimizer.step()
        
        if batch_idx % 100 == 0:
            print(f'Epoch {epoch}, Batch {batch_idx}, Loss: {loss.item() / len(images)}')

    # Logging average loss per epoch
    average_loss = train_loss / len(train_loader.dataset)
    print(f'Epoch {epoch}, Average Loss: {average_loss}')

import matplotlib.pyplot as plt
import numpy as np

# Function to display a set of images
def show_images(images, title_texts):
    fig, axes = plt.subplots(1, len(images), figsize=(20, 4))
    for ax, img, title in zip(axes, images, title_texts):
        ax.imshow(img, cmap='gray')
        ax.set_title(title)
        ax.axis('off')
    plt.show()

# Generate new images conditioned on labels
cvae.eval()  # Make sure the model is in eval mode
num_samples = 10  # Number of samples to generate
condition_labels = ['Cardiomegaly', 'Aortic enlargement']  # Adjust as needed

generated_images = []
with torch.no_grad():
    for label in condition_labels:
        # Create one-hot encoded labels for each condition
        condition_vector = torch.zeros((num_samples, 2))
        condition_vector[:, label_mapping[label]] = 1.0
        condition_vector = condition_vector.to(device)
        
        # Sample random latent vectors
        z = torch.randn(num_samples, 20).to(device)
        
        # Generate images from the latent vectors and conditioned labels
        recon_images = cvae.decoder(z, condition_vector).cpu()
        
        # Move images to CPU and convert to numpy for plotting
        for i in range(num_samples):
            img = recon_images[i].view(128, 128).numpy()  # Assuming images are 128x128
            generated_images.append((img, label))

# Display generated images
show_images([img for img, _ in generated_images], [label for _, label in generated_images])
